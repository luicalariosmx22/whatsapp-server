Python (Cliente)

import socketio
import asyncio

# Crear cliente WebSocket
sio = socketio.AsyncClient()

@sio.event
async def connect():
    print('‚úÖ Conectado al servidor WhatsApp')
    await sio.emit('get_qr', {})

@sio.event
async def qr_code(data):
    print('üì± QR Code recibido:', data)
    
    # Guardar QR como imagen
    if 'qr_code' in data:
        import base64
        qr_data = data['qr_code'].split(',')[1]
        with open('qr_whatsapp.png', 'wb') as f:
            f.write(base64.b64decode(qr_data))

@sio.event
async def whatsapp_status(data):
    print('üìä Estado WhatsApp:', data)

async def main():
    await sio.connect('wss://tu-servidor.railway.app')
    await sio.wait()

if __name__ == '__main__':
    asyncio.run(main())
